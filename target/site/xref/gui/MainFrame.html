70  	private void SearchList(String s) {
71  		DefaultListModel<Item> listModel = new DefaultListModel<Item>();
72  		ArrayList<Item> items = (ArrayList<Item>) Main.getItemList();
73  		for (Item item : items) {
74  			if (item.getName().startsWith(s)
75  					|| item.getName().contains(" " + s))
76  				listModel.addElement(item);
77  		}
78  		if (list == null)
79  			list = new JList<Item>(listModel);
80  		else
81  			list.setModel(listModel);
82  	}

84  	private void refreshList() {
85  		DefaultListModel<Item> listModel = new DefaultListModel<Item>();
86  		ArrayList<Item> items = (ArrayList<Item>) Main.getItemList();
87  		for (Item item : items) {
88  			listModel.addElement(item);
89  		}
90  		if (list == null)
91  			list = new JList<Item>(listModel);
92  		else
93  			list.setModel(listModel);
94  	}

96  	private void refreshTable() throws NullPointerException {
97  		String[] columnNames = new String[] { "Id", "Name", "Quantity",
98  				"Price", "Description" };
99  		Object[][] temp = Main.getTableData();
100 		DefaultTableModel tableModel = new DefaultTableModel(
101 				temp, columnNames);
102 		summa=0;
103 		for (int i=0;i<temp.length;i++){
104 			summa+=(Integer)(temp[i][2])*(Integer)(temp[i][3]);
105 		}
106 		sum.setText(summa.toString());
107 		if (table == null)
108 			table = new JTable(tableModel);
109 		else
110 			table.setModel(tableModel);
111 	}

234 	public void actionPerformed(ActionEvent e) {
235 		if (e.getSource() == addNewItembtn) {
236 			ItemPanel itemPanel = new ItemPanel();
237 			if (ValidatableDialog.show(null, itemPanel, "Add new Item")) {
238 				Main.addNewItem(itemPanel.getItem());
239 				refreshList();
240 			}
241 		}
242 
243 		if (e.getSource() == addToBillbtn) {
244 			try {
245 				Item item = list.getSelectedValue();
246 				if (quantityTextField.getText().isEmpty())
247 					JOptionPane.showMessageDialog(null,
248 							"Quantity can't be empty", "Warning",
249 							JOptionPane.WARNING_MESSAGE);
250 				else {
251 					quantity = Integer.parseInt(quantityTextField.getText());
252 					if (item.getQuantity() > quantity) {
253 						item.changeQuantity(-quantity);
254 						Main.modifyItem(item);
255 						Main.addToTable(item, quantity);
256 						summa += item.getPrice() * quantity;
257 						refreshTable();
258 					} else
259 						JOptionPane.showMessageDialog(null,
260 								"We dont't have enaugh of this!", "Warning",
261 								JOptionPane.WARNING_MESSAGE);
262 				}
263 			} catch (NullPointerException e1) {
264 				JOptionPane.showMessageDialog(null, "No Selected Item",
265 						"Warning", JOptionPane.WARNING_MESSAGE);
266 			}
267 		}
268 		if (e.getSource() == removeFromBill) {
269 			try {
270 
271 				int index = table.getSelectedRow();
272 				if (index == -1)
273 					JOptionPane.showMessageDialog(null, "No Selected Item",
274 							"Warning", JOptionPane.WARNING_MESSAGE);
275 				else {
276 					Object[][] temp = Main.getTableData();
277 					summa -= (Integer) (temp[index][2])
278 							* (Integer) (temp[index][3]);
279 					Main.removeFromTable(index);
280 					refreshTable();
281 				}
282 			} catch (NullPointerException e1) {
283 				JOptionPane.showMessageDialog(null, "No Selected Item",
284 						"Warning", JOptionPane.WARNING_MESSAGE);
285 			}
286 		}
287 
288 		if (e.getSource() == removeItembtn) {
289 			try {
290 				if (JOptionPane.YES_OPTION == JOptionPane.showConfirmDialog(
291 						null, "Are you sure you want to delete the following Item \n" + list.getSelectedValue().toString2(), "Removing Item",
292 						JOptionPane.YES_NO_OPTION)) {
293 					Main.removeItem(list.getSelectedValue());
294 					refreshList();
295 				}
296 			} catch (NullPointerException e1) {
297 				JOptionPane.showMessageDialog(null, "No Selected Item",
298 						"Warning", JOptionPane.WARNING_MESSAGE);
299 			}
300 
301 		}
302 
303 		if (e.getSource() == loadBill) {
304 			
305 			if(!loadBill.getText().equals("OK")){
306 			if (JOptionPane.YES_OPTION == JOptionPane.showConfirmDialog(
307 					null, "Are you sure you want to load another bill? \n The current data will be lost", "Removing Item",
308 					JOptionPane.YES_NO_OPTION)) {
309 			JFileChooser fc = new JFileChooser();
310 			fc.setCurrentDirectory(new File("bills"));
311 			FileNameExtensionFilter xmlfilter = new FileNameExtensionFilter(
312 					"xml files (*.xml)", "xml");
313 			fc.setFileFilter(xmlfilter);
314 			int returnVal = fc.showOpenDialog(null);
315 			if (returnVal == JFileChooser.APPROVE_OPTION) {
316 				String path = fc.getSelectedFile().getAbsolutePath();
317 				Main.loadBill(path);
318 				refreshTable();
319 				loadBill.setText("OK");
320 				saveBill.setEnabled(false);
321 				removeFromBill.setEnabled(false);
322 				addToBillbtn.setEnabled(false);
323 			}
324 		}}else{
325 			loadBill.setText("Load Bill");
326 			saveBill.setEnabled(true);
327 			removeFromBill.setEnabled(true);
328 			addToBillbtn.setEnabled(true);
329 			Main.clearTable();
330 			refreshTable();
331 		}
332 			}
333 			
334 
335 		if (e.getSource() == modifyItembtn) {
336 			try {
337 				ItemPanel itemPanel = new ItemPanel(list.getSelectedValue());
338 				if (ValidatableDialog.show(null, itemPanel, "Modify an Item")) {
339 					Main.modifyItem(itemPanel.getItem());
340 					refreshList();
341 				}
342 			} catch (NullPointerException e1) {
343 				JOptionPane.showMessageDialog(null, "No Selected Item",
344 						"Warning", JOptionPane.WARNING_MESSAGE);
345 			}
346 		}
347 
348 		if (e.getSource() == saveBill) {
349 			Main.saveBill();
350 			Main.clearTable();
351 			refreshTable();
352 		}
353 
354 	}
